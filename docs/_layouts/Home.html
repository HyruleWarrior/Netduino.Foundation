Digital Temperature Sample

Add Netduino.Foundation.TMP36 and .LCD via nuget, and add the following code to a new Netdunio project:

            
using Netduino.Foundation;

public static void Main()
{
var lcdScreen = new LCDScreen (x, y);
 var tempSensor = new AnalogTemperature (pin3);
 var ctoF = new CelsiusToFahrenheitBlock ();
 
 tempSensor.ConnectTo (ctoF);
 ctoF.ConnectTo (lcdScreen);

 while (true) { Thread.Sleep (1000); }
}
